    private void cargarLista() {
            mods = new ArrayList<>();
            mods.addAll(t.getMods());
        DefaultListModel lm = new DefaultListModel();
        for (Modificacion m : mods) {
            lm.addElement("Modificación " + m.getId());
        }
        lstMods.setModel(lm);
    }
